% See:  https://www.overleaf.com/learn/latex/Line_breaks_and_blank_spaces
\clearpage % New page but flushed out figures before the new page
\breakpage % New page but try to fill all remaining space
\newpage % New page or new column if multi column document

\hspace{1cm} % Insert horizontal blank space of 1cm (e.g. between 2 words)
\hfill % Insert blank space to fill the line
\hrulefill % Same as \hfill with a ruler
\dotfill % Same as \hfill with dots

\vspace{1cm} % Insert vertical blank space of 1cm (e.g. between 2 paragraphs)
\vfill % Insert blank space to fill the page
\smallskip % Insert small blank space
\medskip % Insert medium blank space
\bigskip % Insert big blank space

\break % Force to break the line (recommanded to fill the line before)
\linebreak[priority] % Produce the best line break possible
